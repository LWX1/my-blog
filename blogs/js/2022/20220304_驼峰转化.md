## 驼峰转化

```js

function convertUpper(obj) {
	let keyObj = {};
	for(let i in obj) {
		const key = i.replace(/_([a-z])/g, (match, params) => {
			return params.toUpperCase();
		})
		keyObj[key] = obj[i]
	}
	return keyObj
}

const obj = {
	'a_bc_c': 2,
	'cD_gc': 3
}

console.log(convertUpper(obj))

```
